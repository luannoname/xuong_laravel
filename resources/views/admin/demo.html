function previewImageAlbum(input) { const previewContainer = document.getElementById('preview_images'); previewContainer.innerHTML = ''; // Xóa hình ảnh hiện tại if (input.files) { Array.from(input.files).forEach(file => { const reader = new FileReader(); reader.onload = function(e) { // Tạo thẻ img mới và thêm vào container const img = document.createElement('img'); img.src = e.target.result; img.className = 'variant-icon'; previewContainer.appendChild(img); } reader.readAsDataURL(file); }); } }
